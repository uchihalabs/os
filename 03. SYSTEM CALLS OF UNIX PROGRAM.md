# SYSTEM CALLS OF UNIX PROGRAM
```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/types.h>
#include <sys/wait.h>

int main(int argc, char *argv[]) {
    int pid;
    char s[100];

    if (argc < 3) {
        fprintf(stderr, "Usage: %s <file_to_cat> <command_to_exec>\n", argv[0]);
        return 1;
    }

    pid = fork();
    if (pid < 0) {
        perror("Fork failed");
        return 1;
    } else if (pid > 0) {
        // Parent process
        wait(NULL);
        printf("\nParent Process:\n");
        printf("\tParent Process ID: %d\n", getpid());
        
        execlp("cat", "cat", argv[1], (char*)0);
        perror("Can't execute cat");
    } else {
        // Child process
        printf("\nChild Process:\n");
        printf("\tChild's Parent ID: %d\n", getppid());
        printf("\tChild Process ID: %d\n", getpid());
        
        execvp(argv[2], &argv[2]);
        perror("Can't execute command");
    }

    return 0;
}
```
# Output
```
Child process executing
<contents of example.txt>
Parent process completed
```
