```c
#include<stdio.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/wait.h>

int main(int argc, char *argv[]) {
    pid_t pid;
    char *command[] = {"cat", argv[1], NULL};

    pid = fork();

    if (pid < 0) {
        perror("Fork failed");
        return 1;
    } else if (pid == 0) {
        printf("Child process executing\n");
        execvp(command[0], command);
        perror("Exec failed");
        return 1;
    } else {
        wait(NULL);
        printf("Parent process completed\n");
    }

    return 0;
}
```
