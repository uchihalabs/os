```c
#include<stdio.h>
#include<stdlib.h>

int main() {
    int n, t[20], tot = 0;
    printf("Enter the number of tracks: ");
    scanf("%d", &n);

    printf("Enter the track positions: ");
    for (int i = 0; i < n; i++) {
        scanf("%d", &t[i]);
    }

    for (int i = 1; i < n; i++) {
        tot += abs(t[i] - t[i - 1]);
    }

    printf("Total head movement: %d\n", tot);
    printf("Average head movement: %.2f\n", tot / (float)(n - 1));
    return 0;
}
```
