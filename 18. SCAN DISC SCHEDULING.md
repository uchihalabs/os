#include<stdio.h>
#include<stdlib.h>

int main() {
    int n, t[20], head, tot = 0;
    printf("Enter the number of tracks: ");
    scanf("%d", &n);

    printf("Enter the track positions: ");
    for (int i = 0; i < n; i++) {
        scanf("%d", &t[i]);
    }

    printf("Enter the head position: ");
    scanf("%d", &head);

    // Sorting track positions
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (t[i] > t[j]) {
                int temp = t[i];
                t[i] = t[j];
                t[j] = temp;
            }
        }
    }

    for (int i = 0; i < n; i++) {
        if (t[i] >= head) {
            tot += abs(head - t[i]);
            head = t[i];
        }
    }
    for (int i = n - 1; i >= 0; i--) {
        if (t[i] < head) {
            tot += abs(head - t[i]);
            head = t[i];
        }
    }

    printf("Total head movement: %d\n", tot);
    return 0;
}
